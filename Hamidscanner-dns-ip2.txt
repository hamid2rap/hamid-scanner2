#!/bin/bash

RED="\e[31m"; GREEN="\e[32m"; YELLOW="\e[33m"; BLUE="\e[34m"; CYAN="\e[36m"; RESET="\e[0m"

echo -e "${CYAN}=== HAMID Cloudflare & DNS Turbo Scanner ===${RESET}"
echo -e "${YELLOW}Fast scanning + progress percentage (Termux compatible)${RESET}"
echo

echo -e "${BLUE}1) Scan Cloudflare IP ranges"
echo -e "2) Turbo DNS scan (first 1000 DNS from public-dns.info)"
echo -e "3) Exit${RESET}"
echo
read -p "Select option: " OPT

############################################
# OPTION 2 — TURBO DNS SCANNER (1000 DNS)
############################################
if [[ "$OPT" == "2" ]]; then
    echo
    echo -e "${CYAN}Fetching 1000 DNS from public-dns.info...${RESET}"

    DNS_URL="https://public-dns.info/nameservers.txt"

    # Save DNS list to a temp file (prevents curl error 23)
    DNS_FILE=$(mktemp)

    curl -A "Mozilla/5.0" -fsSL "$DNS_URL" -o "$DNS_FILE"

    if [[ ! -s "$DNS_FILE" ]]; then
        echo -e "${RED}Failed to fetch DNS list.${RESET}"
        exit 1
    fi

    # Read ONLY first 1000 DNS lines
    DNS_ARRAY=()
    COUNT=0
    while read -r line; do
        [[ -z "$line" ]] && continue
        DNS_ARRAY+=("$line")
        COUNT=$((COUNT + 1))
        [[ $COUNT -ge 1000 ]] && break
    done < "$DNS_FILE"

    TOTAL=${#DNS_ARRAY[@]}
    DONE=0
    OUT_DNS=$(mktemp)

    echo -e "${BLUE}Scanning $TOTAL DNS servers...${RESET}"

    scan_dns() {
        DNS=$1
        PING=$(ping -c 1 -W 1 "$DNS" 2>/dev/null | awk -F'=' '/time=/ {print $4}' | cut -d' ' -f1)

        if [[ -n "$PING" ]]; then
            echo "$DNS $PING" >> "$OUT_DNS"
        fi

        DONE=$((DONE + 1))
        PERCENT=$((DONE * 100 / TOTAL))
        printf "${GREEN}Progress: %d%%%s\r" "$PERCENT" "$RESET"
    }

    export -f scan_dns
    export OUT_DNS DONE TOTAL

    # 40 threads = FAST & stable for Termux
    printf "%s\n" "${DNS_ARRAY[@]}" | xargs -n1 -P40 bash -c 'scan_dns "$@"' _

    echo
    echo -e "${GREEN}Selecting top 10 DNS servers...${RESET}"
    echo

    BEST=$(sort -k2 -n "$OUT_DNS" | head -n 10)

    declare -A PAIRS

    while read -r DNS PING; do
        BASE=$(echo "$DNS" | awk -F'.' '{print $1"."$2"."$3}')
        PAIRS["$BASE"]+="$DNS "
    done <<< "$BEST"

    n=1
    for BASE in "${!PAIRS[@]}"; do
        echo -e "${CYAN}$n) ${GREEN}${PAIRS[$BASE]}${RESET}"
        n=$((n+1))
    done

    echo
    echo -e "${MAGENTA}DNS Turbo scan complete — HAMID Scanner${RESET}"
    exit 0
fi

############################################
# OPTION 1 — CLOUDFLARE SCANNER
############################################

if [[ "$OPT" == "1" ]]; then

RANGES=(
"104.16.0.0 104.23.255.255"
"104.24.0.0 104.27.255.255"
"172.64.0.0 172.71.255.255"
"162.158.0.0 162.159.255.255"
"198.41.128.0 198.41.255.255"
"188.114.96.0 188.114.111.255"
"190.93.240.0 190.93.255.255"
)

ip2num() {
    IFS=. read -r a b c d <<< "$1"
    echo $((a*256**3 + b*256**2 + c*256 + d))
}

num2ip() {
    local ip=$1
    echo "$((ip/256**3)).$((ip/256**2%256)).$((ip/256%256)).$((ip%256))"
}

OUT=$(mktemp)

TOTAL=0
for RANGE in "${RANGES[@]}"; do
    read S E <<< "$RANGE"
    S=$(ip2num "$S")
    E=$(ip2num "$E")
    TOTAL=$((TOTAL + ( (E-S)/256 )))
done

DONE=0

echo -e "${BLUE}Scanning Cloudflare ranges...${RESET}"

for RANGE in "${RANGES[@]}"; do
    read START END <<< "$RANGE"
    S=$(ip2num "$START")
    E=$(ip2num "$END")

    echo -e "${CYAN}Range: $START → $END${RESET}"

    for ((IP=S; IP<=E; IP+=256)); do
        TARGET=$(num2ip "$IP")

        (
            ping -c 1 -W 1 "$TARGET" >/dev/null 2>&1
            if [ $? -eq 0 ]; then
                PING=$(ping -c 1 -W 1 "$TARGET" 2>/dev/null | awk -F'=' '/time=/ {print $4}' | cut -d' ' -f1)
                [[ -n "$PING" ]] && echo "$TARGET $PING" >> "$OUT"
            fi
        ) &

        DONE=$((DONE + 1))
        PERCENT=$((DONE * 100 / TOTAL))

        printf "${GREEN}Progress: %d%%%s\r" "$PERCENT" "$RESET"
    done
done

wait

echo
echo -e "${GREEN}Selecting top 10 Cloudflare IPs...${RESET}"
echo

sort -k2 -n "$OUT" | head -n 10 | nl | while read n ip ping; do
    echo -e "${CYAN}$n) ${GREEN}$ip${RESET}  →  ${YELLOW}${ping} ms${RESET}"
done

echo
echo -e "${MAGENTA}Cloudflare scan complete — HAMID Scanner${RESET}"

fi

############################################
# OPTION 3 — EXIT
############################################
if [[ "$OPT" == "3" ]]; then
    echo -e "${RED}Exiting...${RESET}"
    exit 0
fi
