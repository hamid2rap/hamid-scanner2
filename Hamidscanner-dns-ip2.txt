#!/bin/bash

RED="\e[31m"; GREEN="\e[32m"; YELLOW="\e[33m"; BLUE="\e[34m"; CYAN="\e[36m"; RESET="\e[0m"

echo -e "${CYAN}=== HAMID Cloudflare Scanner (Ultra Fast) ===${RESET}"
echo -e "${YELLOW}اسکن ۷ رنج کلودفلر با درصد پیشرفت و انتخاب ۱۰ آی‌پی برتر...${RESET}"
echo

# Cloudflare ranges (start-end)
RANGES=(
"104.16.0.0 104.23.255.255"
"104.24.0.0 104.27.255.255"
"172.64.0.0 172.71.255.255"
"162.158.0.0 162.159.255.255"
"198.41.128.0 198.41.255.255"
"188.114.96.0 188.114.111.255"
"190.93.240.0 190.93.255.255"
)

# Convert IP → number
ip2num() {
    IFS=. read -r a b c d <<< "$1"
    echo $((a*256**3 + b*256**2 + c*256 + d))
}

# Convert number → IP
num2ip() {
    local ip=$1
    echo "$((ip/256**3)).$((ip/256**2%256)).$((ip/256%256)).$((ip%256))"
}

OUT="hamid_cf_scan.txt"
> "$OUT"

TOTAL=0
for RANGE in "${RANGES[@]}"; do
    read S E <<< "$RANGE"
    S=$(ip2num "$S")
    E=$(ip2num "$E")
    COUNT=$(( (E-S) / 256 ))
    TOTAL=$((TOTAL + COUNT))
done

DONE=0

echo -e "${BLUE}در حال اسکن...${RESET}"

for RANGE in "${RANGES[@]}"; do
    read START END <<< "$RANGE"
    S=$(ip2num "$START")
    E=$(ip2num "$END")

    echo -e "${CYAN}رنج: $START → $END${RESET}"

    for ((IP=S; IP<=E; IP+=256)); do
        TARGET=$(num2ip "$IP")

        (
            ping -c 1 -W 1 "$TARGET" >/dev/null 2>&1
            if [ $? -eq 0 ]; then
                PING=$(ping -c 1 -W 1 "$TARGET" 2>/dev/null | awk -F'=' '/time=/ {print $4}' | cut -d' ' -f1)
                if [[ -n "$PING" ]]; then
                    echo "$TARGET $PING" >> "$OUT"
                fi
            fi
        ) &

        DONE=$((DONE + 1))
        PERCENT=$((DONE * 100 / TOTAL))
        echo -ne "${GREEN}پیشرفت: ${PERCENT}%${RESET}\r"
    done
done

wait

echo
echo -e "${GREEN}در حال انتخاب ۱۰ آی‌پی برتر...${RESET}"
echo

sort -k2 -n "$OUT" | head -n 10 | nl | while read n ip ping; do
    echo -e "${CYAN}$n) ${GREEN}$ip${RESET}  →  ${YELLOW}${ping} ms${RESET}"
done

echo
echo -e "${MAGENTA}اسکن تمام شد — HAMID Cloudflare Scanner${RESET}"
