)

TOTAL=${#IPS[@]}
DONE=0

OUT=$(mktemp)

echo -e "${BLUE}در حال اسکن ${TOTAL} آی‌پی واقعی کلودفلر...${RESET}"

scan_ip() {
    IP=$1
    PING=$(ping -c 1 -W 1 "$IP" 2>/dev/null | awk -F'=' '/time=/ {print $4}' | cut -d' ' -f1)

    if [[ -n "$PING" ]]; then
        echo "$IP $PING" >> "$OUT"
    fi

    DONE=$((DONE + 1))
    PERCENT=$((DONE * 100 / TOTAL))
    echo -ne "${GREEN}پیشرفت: ${PERCENT}%${RESET}\r"
}

export -f scan_ip
export OUT DONE TOTAL

printf "%s\n" "${IPS[@]}" | xargs -n1 -P20 bash -c 'scan_ip "$@"' _

echo
echo -e "${GREEN}در حال انتخاب ۱۰ آی‌پی برتر...${RESET}"
echo

sort -k2 -n "$OUT" | head -n 10 | nl | while read n ip ping; do
    echo -e "${CYAN}$n) ${GREEN}$ip${RESET}  →  ${YELLOW}${ping} ms${RESET}"
done

echo
