#!/bin/bash

# ============================
#      HAMID Warp Scanner
# ============================

RED="\e[31m"
GREEN="\e[32m"
YELLOW="\e[33m"
BLUE="\e[34m"
MAGENTA="\e[35m"
CYAN="\e[36m"
RESET="\e[0m"

banner() {
    echo -e "${MAGENTA}"
    echo "██╗  ██╗ █████╗ ███╗   ███╗██╗██╗██████╗ "
    echo "██║  ██║██╔══██╗████╗ ████║██║██║██╔══██╗"
    echo "███████║███████║██╔████╔██║██║██║██║  ██║"
    echo "██╔══██║██╔══██║██║╚██╔╝██║██║██║██║  ██║"
    echo "██║  ██║██║  ██║██║ ╚═╝ ██║██║██║██████╔╝"
    echo "╚═╝  ╚═╝╚═╝  ╚═╝╚═╝     ╚═╝╚═╝╚═╝╚═════╝ "
    echo -e "${CYAN}           HAMID Warp Scanner${RESET}"
    echo
}

menu() {
    echo -e "${BLUE}1) اجرای اسکن سریع"
    echo -e "2) اجرای اسکن معمولی"
    echo -e "3) اجرای اسکن عمیق"
    echo -e "4) اجرای اسکن سفارشی"
    echo -e "5) خروج${RESET}"
    echo
    echo -ne "${YELLOW}گزینه را انتخاب کن: ${RESET}"
    read CH
}

banner

OS=$(uname -s)
if [ "$OS" != "Linux" ]; then
    echo -e "${RED}این اسکریپت فقط روی Linux/Android اجرا می‌شود.${RESET}"
    exit 1
fi

ARCH=$(uname -m)
case "$ARCH" in
    aarch64|arm64) ARCH="arm64" ;;
    armv7*|armv8*) ARCH="arm" ;;
    x86_64)        ARCH="amd64" ;;
    i386|i686)     ARCH="386" ;;
    *)             echo -e "${RED}معماری پشتیبانی نمی‌شود: ${ARCH}${RESET}" && exit 1 ;;
esac

BINARY="BPB-Warp-Scanner"
ARCHIVE="${BINARY}-linux-${ARCH}.tar.gz"
LATEST_VERSION=$(curl -fsSL https://raw.githubusercontent.com/bia-pain-bache/BPB-Warp-Scanner/main/VERSION)

menu

case "$CH" in
    1) MODE="100" ;;
    2) MODE="1000" ;;
    3) MODE="10000" ;;
    4)
        echo -ne "${CYAN}چند endpoint؟ ${RESET}"
        read MODE
        ;;
    5)
        echo -e "${RED}خروج...${RESET}"
        exit 0
        ;;
    *)
        echo -e "${RED}گزینه نامعتبر!${RESET}"
        exit 1
        ;;
esac

if [ ! -x "./${BINARY}" ]; then
    echo -e "${YELLOW}اسکنر پیدا نشد، در حال نصب...${RESET}"
    curl -L -# -o "${ARCHIVE}" \
        "https://github.com/bia-pain-bache/BPB-Warp-Scanner/releases/latest/download/${ARCHIVE}"
    tar xzf "./${ARCHIVE}"
fi

echo -e "${GREEN}در حال اجرای HAMID Warp Scanner...${RESET}"
echo -e "${BLUE}Mode: ${MODE} endpoints${RESET}"
echo

# اجرای اسکنر بدون نمایش خروجی BPB
OUTPUT=$("./${BINARY}" --scan "${MODE}" 2>/dev/null | sed '/BPB/d')

echo -e "${CYAN}نتیجه اسکن:${RESET}"
echo -e "${GREEN}${OUTPUT}${RESET}"
