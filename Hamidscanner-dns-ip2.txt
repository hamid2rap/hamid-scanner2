#!/usr/bin/env bash

TOTAL=2000
RESULTS="$(mktemp)"
COUNT=0

# تعداد پینگ‌های همزمان
PARALLEL=40

RANGES=(
  "104.16.0.0/12"
  "104.24.0.0/14"
  "172.64.0.0/13"
  "141.101.64.0/18"
  "188.114.96.0/20"
  "190.93.240.0/20"
  "198.41.128.0/17"
  "103.21.244.0/22"
  "103.22.200.0/22"
  "103.31.4.0/22"
)

cidr_random_ip() {
  local cidr="$1"
  local base="${cidr%/*}"
  local prefix="${cidr#*/}"

  IFS='.' read -r o1 o2 o3 o4 <<< "$base"
  local ip_num=$(( (o1<<24) + (o2<<16) + (o3<<8) + o4 ))

  local host_bits=$((32 - prefix))
  local block_size=$((1 << host_bits))

  local offset=$(( RANDOM % block_size ))
  local cur=$(( ip_num + offset ))

  printf "%d.%d.%d.%d\n" \
    $(( (cur>>24)&255 )) \
    $(( (cur>>16)&255 )) \
    $(( (cur>>8)&255 )) \
    $(( cur&255 ))
}

run_ping() {
  local ip="$1"
  ping_out=$(ping -c 1 -W 1 "$ip" 2>/dev/null)
  if [[ $? -eq 0 ]]; then
    avg=$(echo "$ping_out" | awk -F'/' '/rtt/ {print $5}')
    [[ -n "$avg" ]] && echo "$avg $ip" >> "$RESULTS"
  fi
}

echo "[*] شروع تست فوق‌سریع ۲۰۰۰ آی‌پی..."
echo

while (( COUNT < TOTAL )); do
  for ((i=0; i<PARALLEL && COUNT<TOTAL; i++)); do
    range="${RANGES[$RANDOM % ${#RANGES[@]}]}"
    ip=$(cidr_random_ip "$range")

    run_ping "$ip" &

    COUNT=$((COUNT+1))
    PERCENT=$(( COUNT * 100 / TOTAL ))
    echo -ne "درصد پیشرفت: $PERCENT% \r"
  done

  wait
done

echo
echo "[*] انتخاب ۱۰ آی‌پی برتر..."
echo

if [[ -s "$RESULTS" ]]; then
  sort -n "$RESULTS" | head -n 10 | awk '{print "IP: "$2"   RTT: "$1" ms"}'
else
  echo "[!] هیچ آی‌پی سالم پیدا نشد."
fi

rm -f "$RESULTS"
