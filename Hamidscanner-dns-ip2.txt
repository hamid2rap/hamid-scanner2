#!/data/data/com.termux/files/usr/bin/bash

# Colors
A='\033[0;32m';C='\033[1;34m';Z='\033[0m';O='\033[0;33m'
RED="\033[1;31m";GREEN="\033[1;32m";YELLOW="\033[1;33m"
BLUE="\033[1;34m";MAGENTA="\033[1;35m";CYAN="\033[1;36m";RESET="\033[0m"

# Menu
echo -e "${C}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${Z}"
echo -e "${C} Hamid MultiScanner™ Menu ${Z}"
echo -e "${C}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${Z}"
echo "1) IP Scan"
echo "2) DNS Speed Test"
echo "0) Exit"
read -p "Choice: " choice

case $choice in
  1)
    ########################################
    # IP Scan
    ########################################
    RESULTS=()
    Y=0
    TOTAL=1500
    echo -e "\n${C}🌐 HamidScanner™ ($TOTAL IPs, progress counter + percent)${Z}\n"
    TMP=$(mktemp)

    scan_ip() {
      local id=$1
      local ip=$2
      P=$(ping -c1 -W1 "$ip" 2>/dev/null)
      M=$(echo "$P" | grep time= | awk -F'time=' '{print $2}' | awk '{print $1}')
      if [[ -n "$M" ]]; then
        echo "$M $ip" >> "$TMP"
      fi
    }

    while(( Y<TOTAL ));do
      ((Y++))
      case $((RANDOM%6)) in
        0)H="172.65.$((RANDOM%256)).$((RANDOM%256))";;
        1)H="162.159.$((RANDOM%256)).$((RANDOM%256))";;
        2)H="104.18.$((RANDOM%256)).$((RANDOM%256))";;
        3)H="104.26.$((RANDOM%256)).$((RANDOM%256))";;
        4)H="172.67.$((RANDOM%256)).$((RANDOM%256))";;
        5)H="172.68.$((RANDOM%256)).$((RANDOM%256))";;
      esac

      PERCENT=$(( Y*100/TOTAL ))
      echo -e "${A}HamidScanner #$Y/$TOTAL → $PERCENT%${Z}"

      scan_ip "$Y" "$H" &

      if (( Y % 50 == 0 )); then
        wait
      fi
    done

    wait
    BEST=$(sort -n "$TMP" | head -n 5)

    echo -e "\n${O}╔══════════════════════════════════════════════════════════════════════╗${Z}"
    echo -e "${O}║${Z}                        Made by @hamid2rap                           ${O}║${Z}"
    echo -e "${O}╚══════════════════════════════════════════════════════════════════════╝${Z}"

    echo -e "\n${C}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    echo -e "🎯 HamidScanner™ Best 5 Results:"
    while read -r line; do
      M=$(echo $line | awk '{print $1}')
      H=$(echo $line | awk '{print $2}')
      echo -e "${A}✅ $H → ${M}ms${Z}"
    done <<< "$BEST"
    echo -e "${C}📦 HamidScanner™${Z}"
    echo -e "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n"
    rm -f "$TMP"
    ;;
  2)
    ########################################
    # DNS Speed Test
    ########################################
    echo -e "${MAGENTA}"
    echo "╔════════════════════════════════════════════════════════════════════════════════════╗"
    echo "║                               Made by Hamid @hamid2rap                             ║"
    echo "╚════════════════════════════════════════════════════════════════════════════════════╝"
    echo -e "${RESET}"

    echo -e "${YELLOW}Fetching public DNS list from public-dns.info...${RESET}"
    curl -s https://public-dns.info/nameservers.csv -o d.tmp
    X=$(awk -F, 'NR>1{print $1}' d.tmp | head -n 100)
    declare -A T

    echo -e "${YELLOW}Checking DNS servers...${RESET}"
    printf "  ${MAGENTA}%-18s %-10s${RESET}\n" "Server" "Ping"

    for H in $X; do
      R=$(ping -c1 -W1 -q "$H" | awk -F/ '/rtt/{print $5}')
      if [[ $R ]]; then
        printf "  ${CYAN}%-18s${RESET} → ${GREEN}%6s ms${RESET}\n" "$H" "$R"
        T[$H]=$R
      else
        printf "  ${RED}%-18s → No reply${RESET}\n" "$H"
      fi
    done

    get_pair() {
      case "$1" in
        "8.8.8.8") echo "8.8.4.4" ;;
        "1.1.1.1") echo "1.0.0.1" ;;
        "9.9.9.9") echo "149.112.112.112" ;;
        "208.67.222.222") echo "208.67.220.220" ;;
        *) echo "" ;;
      esac
    }

    echo -e "\n${GREEN}Fastest 3 (with pairs):${RESET}"
    printf "  ${MAGENTA}%-18s %-10s %-30s${RESET}\n" "Server" "Ping" "Note"

    TOP3=$(for K in "${!T[@]}"; do
      echo "$K ${T[$K]}"
    done | sort -k2 -n | head -n 3)

    FINAL_LIST=""
    echo "$TOP3" | while read -r S P; do
      printf "  ${CYAN}%-18s${RESET} → ${GREEN}%6s ms${RESET}   ${MAGENTA}@hamid2rap${RESET}\n" "$S" "$P"
      FINAL_LIST+="nameserver $S\n"
      PAIR=$(get_pair "$S")
      if [[ -n "$PAIR" ]]; then
        printf "  ${CYAN}%-18s${RESET} → ${YELLOW}paired${RESET}\n" "$PAIR"
        FINAL_LIST+="nameserver $PAIR\n"
      fi
    done

    echo -e "$FINAL_LIST" > best-dns.conf
    echo -e "\n${BLUE}Saved fastest DNS servers (with pairs) to best-dns.conf${RESET}"
    ;;
  0)
    echo "Exit..."
    ;;
  *)
    echo "Invalid choice!"
    ;;
esac
