#!/bin/bash

RED="\e[31m"; GREEN="\e[32m"; YELLOW="\e[33m"; BLUE="\e[34m"; CYAN="\e[36m"; RESET="\e[0m"

echo -e "${CYAN}=== HAMID Cloudflare IP Scanner ===${RESET}"
echo -e "${YELLOW}Fast Cloudflare scanning + progress percentage${RESET}"
echo

echo -e "${BLUE}1) Scan Cloudflare IP ranges"
echo -e "2) Exit${RESET}"
echo
read -p "Select option: " OPT

############################################
# OPTION 1 — CLOUDFLARE IP SCANNER
############################################
if [[ "$OPT" == "1" ]]; then

    echo
    echo -e "${CYAN}Loading Cloudflare IP ranges...${RESET}"

    # Cloudflare IP ranges (بدون دانلود، داخلی)
    RANGES=(
        "104.16.0.0 104.23.255.255"
        "104.24.0.0 104.27.255.255"
        "172.64.0.0 172.71.255.255"
        "162.158.0.0 162.159.255.255"
        "198.41.128.0 198.41.255.255"
        "188.114.96.0 188.114.111.255"
        "190.93.240.0 190.93.255.255"
    )

    ip2num() {
        IFS=. read -r a b c d <<< "$1"
        echo $((a*256**3 + b*256**2 + c*256 + d))
    }

    num2ip() {
        local ip=$1
        echo "$((ip/256**3)).$((ip/256**2%256)).$((ip/256%256)).$((ip%256))"
    }

    OUT=$(mktemp)

    TOTAL=0
    for RANGE in "${RANGES[@]}"; do
        read S E <<< "$RANGE"
        S=$(ip2num "$S")
        E=$(ip2num "$E")
        TOTAL=$((TOTAL + ( (E-S)/256 )))
    done

    DONE=0

    echo -e "${BLUE}Scanning Cloudflare ranges...${RESET}"

    for RANGE in "${RANGES[@]}"; do
        read START END <<< "$RANGE"
        S=$(ip2num "$START")
        E=$(ip2num "$END")

        echo -e "${CYAN}Range: $START → $END${RESET}"

        for ((IP=S; IP<=E; IP+=256)); do
            TARGET=$(num2ip "$IP")

            (
                ping -c 1 -W 1 "$TARGET" >/dev/null 2>&1
                if [ $? -eq 0 ]; then
                    PING=$(ping -c 1 -W 1 "$TARGET" 2>/dev/null | awk -F'=' '/time=/ {print $4}' | cut -d' ' -f1)
                    [[ -n "$PING" ]] && echo "$TARGET $PING" >> "$OUT"
                fi
            ) &

            DONE=$((DONE + 1))
            PERCENT=$((DONE * 100 / TOTAL))

            printf "${GREEN}Progress: %d%%%s\r" "$PERCENT" "$RESET"
        done
    done

    wait

    echo
    echo -e "${GREEN}Top 10 Cloudflare IPs:${RESET}"
    echo

    sort -k2 -n "$OUT" | head -n 10 | nl

    echo
    echo -e "${MAGENTA}Cloudflare scan complete — HAMID Scanner${RESET}"

fi

############################################
# OPTION 2 — EXIT
############################################
if [[ "$OPT" == "2" ]]; then
    echo -e "${RED}Exiting...${RESET}"
    exit 0
fi
