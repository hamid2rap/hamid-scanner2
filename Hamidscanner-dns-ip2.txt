#!/data/data/com.termux/files/usr/bin/bash

A='\033[0;32m';C='\033[1;34m';Z='\033[0m'
echo -e "${C}🚀 HamidScanner UltraTurbo™ (AutoInstaller)${Z}"

########################################
# 1) Auto Install fping
########################################
if ! command -v fping >/dev/null 2>&1; then
  echo -e "${A}📦 Installing fping...${Z}"
  pkg update -y >/dev/null 2>&1
  pkg install fping -y >/dev/null 2>&1
  echo -e "${A}✔ fping installed${Z}"
fi

########################################
# 2) Generate IP List
########################################
TOTAL=1500
echo -e "${A}🧩 Generating $TOTAL IPs...${Z}"

generate_ips() {
  for ((i=0;i<TOTAL;i++)); do
    case $((RANDOM%6)) in
      0)echo "172.65.$((RANDOM%256)).$((RANDOM%256))";;
      1)echo "162.159.$((RANDOM%256)).$((RANDOM%256))";;
      2)echo "104.18.$((RANDOM%256)).$((RANDOM%256))";;
      3)echo "104.26.$((RANDOM%256)).$((RANDOM%256))";;
      4)echo "172.67.$((RANDOM%256)).$((RANDOM%256))";;
      5)echo "172.68.$((RANDOM%256)).$((RANDOM%256))";;
    esac
  done
}

generate_ips > iplist.txt

########################################
# 3) UltraTurbo Scan
########################################
echo -e "${A}⚡ Starting UltraTurbo Scan...${Z}"

TMP=$(mktemp)

# اجرای fping با سرعت بسیار بالا
fping -c1 -t100 -q -f iplist.txt 2>&1 | \
grep 'avg' | awk '{print $1, $8}' | sed 's/ms,//' > "$TMP"

########################################
# 4) Show Best Results
########################################
echo -e "${C}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${Z}"
echo -e "${A}🎯 Best 5 Results:${Z}"

sort -k2 -n "$TMP" | head -n 5 | while read ip ms; do
  echo -e "${A}✔ $ip → ${ms}ms${Z}"
done

echo -e "${C}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${Z}"
echo -e "${A}Made by Hamid @hamid2rap${Z}"

rm -f "$TMP"
