#!/bin/bash

CYAN="\e[36m"; GREEN="\e[32m"; RESET="\e[0m"

echo -e "${CYAN}=== HAMID Cloudflare FAST Scanner (100 IP per range) ===${RESET}"

# رنج‌های کلودفلر
RANGES=(
    "104.16.0.0 104.23.255.255"
    "104.24.0.0 104.27.255.255"
    "172.64.0.0 172.71.255.255"
    "162.158.0.0 162.159.255.255"
    "198.41.128.0 198.41.255.255"
    "188.114.96.0 188.114.111.255"
    "190.93.240.0 190.93.255.255"
)

ip2num() {
    IFS=. read -r a b c d <<< "$1"
    echo $((a*256**3 + b*256**2 + c*256 + d))
}

num2ip() {
    local ip=$1
    echo "$((ip/256**3)).$((ip/256**2%256)).$((ip/256%256)).$((ip%256))"
}

OUT=$(mktemp)

echo -e "${CYAN}Scanning Cloudflare ranges (100 IP per range)...${RESET}"

scan_ip() {
    IP=$1
    PING=$(ping -c 1 -W 1 "$IP" 2>/dev/null | awk -F'=' '/time=/ {print $4}' | cut -d' ' -f1)
    [[ -n "$PING" ]] && echo "$IP $PING" >> "$OUT"
}

export -f scan_ip
export OUT

for RANGE in "${RANGES[@]}"; do
    read START END <<< "$RANGE"
    S=$(ip2num "$START")
    E=$(ip2num "$END")

    TOTAL_RANGE=$((E - S))
    STEP=$((TOTAL_RANGE / 100))   # فقط ۱۰۰ آی‌پی از هر رنج

    echo -e "${GREEN}Range: $START → $END${RESET}"

    for ((i=0; i<100; i++)); do
        IP_NUM=$((S + STEP * i))
        TARGET=$(num2ip "$IP_NUM")
        echo "$TARGET"
    done

done | xargs -n1 -P40 bash -c 'scan_ip "$@"' _

echo
echo -e "${GREEN}Top Cloudflare IPs:${RESET}"
sort -k2 -n "$OUT" | head -n 10 | nl
